In[17]:
results = {}

for name, model in models.items():
    # Wrap the model in MultiOutputRegressor for multi-output regression
    multioutput_model = MultiOutputRegressor(model)
    
    # Fit the wrapped model to the data
    multioutput_model.fit(X_encoded, Y)
    
    # Predict both Min_Price and Max_Price
    y_pred = multioutput_model.predict(X_encoded)
    
    # Calculate R^2 Score
    r2_score = multioutput_model.score(X_encoded, Y)
    
    # Calculate MSE
    mse = mean_squared_error(Y, y_pred)
    
    # Store results
    results[name] = {
        'R^2 Score': r2_score,
        'MSE': mse
    }
    
# Convert results dictionary to DataFrame
results_df = pd.DataFrame(results).T
